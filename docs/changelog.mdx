<Version 
  version="0.3.0.0-beta" 
  date="5th February, 2026" 
  releaseDate="2026-02-05"
  isLatest={true}
>
  <ChangeType type="added">
    <ChangeItem>
      Extended fontWeight support with string and numeric values.
      <ul>
        <li>String values: thin, extralight or ultralight, light, regular, medium, semibold or semi-bold, bold, extrabold or ultrabold, black</li>
        <li>Numeric values: 100, 200, 300, 400, 500, 600, 700, 800, 900</li>
      </ul>
    </ChangeItem>
    <ChangeItem>Added fontPath property for loading custom fonts in the text element</ChangeItem>
    <ChangeItem>Added inline style tag support for the text element</ChangeItem>
    <ChangeItem>Added fontColor support for gradient colors</ChangeItem>
    <ChangeItem>Added fontShadow support for text</ChangeItem>
    <ChangeItem>Added letterSpacing property for the text element</ChangeItem>
    <ChangeItem>Added case property for the text element</ChangeItem>
    <ChangeItem>Added underline property for the text element</ChangeItem>
    <ChangeItem>Added strikethrough property for the text element</ChangeItem>
  </ChangeType>

  <ChangeType type="added">
    <ChangeItem>
      <strong>Supported Inline Tags:</strong>
      <ul>
        <li><code>{`<b>`}</code> : Bold text</li>
        <li><code>{`<i>`}</code> : Italic text</li>
        <li><code>{`<u>`}</code> : Underlined text</li>
        <li><code>{`<s>`}</code> : Strikethrough text</li>
        <li>
          <code>{`<color=value>`}</code> : Changes text color or applies a gradient
          <ul>
            <li>Example: <code>{`<color=#f00>Red Text</color>`}</code></li>
            <li>Example: <code>{`<color=linearGradient(0,#f00,#0f0)>Gradient Text</color>`}</code></li>
          </ul>
        </li>
        <li>
          <code>{`<size=value>`}</code> : Changes font size in pixels
          <ul>
            <li>Example: <code>{`<size=24>Large Text</size>`}</code></li>
          </ul>
        </li>
        <li>
          <code>{`<font=name>`}</code> : Changes font face
          <ul>
            <li>Example: <code>{`<font=Consolas>Monospaced Text</font>`}</code></li>
          </ul>
        </li>
        <li>
          <code>{`<case=value>`}</code> : Changes text case
          <ul>
            <li><code>normal</code> : Maintains original casing</li>
            <li><code>upper</code> : Converts all text to uppercase</li>
            <li><code>lower</code> : Converts all text to lowercase</li>
            <li><code>capitalize</code> : Capitalizes the first letter of each word</li>
            <li><code>sentence</code> : Capitalizes the first letter of the first word</li>
            <li>Example: <code>{`<case=upper>this will be uppercase</case>`}</code></li>
          </ul>
        </li>
      </ul>
    </ChangeItem>
  </ChangeType>
</Version>

<Version

    <ChangeItem>Updated the installer and uninstaller to handle user data cleanup in AppData.</ChangeItem>
    <ChangeItem>Consolidated resource extraction logic across the codebase for improved performance (caching).</ChangeItem>
  </ChangeType>
