title: Configuration Methods

import CodeBlock from '@theme/CodeBlock';
import { Method, MethodParam } from '@site/src/components/Method';
import Admonition from '@theme/Admonition';

Novadesk provides several methods to control global application behavior programmatically via the `app` object. These methods can be called at any time and their values are
persisted to `settings.json` located in the application's AppData directory.

<Admonition type="warning">
Configuration methods are **only available in the Main script**. UI scripts cannot modify global application settings.
</Admonition>

<Method
  signature="app.saveLogToFile(bool)"
  description="Enables or disables logging to a file (`logs.log`) in the application's AppData directory."
>
  ### Parameters
  <MethodParam name="bool" type="boolean">
    `true` to enable file logging, `false` to disable.
  </MethodParam>
  
  ### Examples
  <CodeBlock language="javascript">
    {`// Enable logging to file
app.saveLogToFile(true);`}
  </CodeBlock>
</Method>
<Method
  signature="app.enableDebugging(bool)"
  description="Sets the global log level. When enabled, debug-level messages will be visible in the console and log file."
>
  ### Parameters
  <MethodParam name="bool" type="boolean">
    `true` to enable debug logging, `false` to use standard informational logging.
  </MethodParam>
  
  ### Examples
  <CodeBlock language="javascript">
    {`app.enableDebugging(true);
console.debug("Detailed diagnostic information");`}
  </CodeBlock>
</Method>
<Method
  signature="app.disableLogging(bool)"
  description="Completely disables or enables all logging output (both console and file)."
>
  ### Parameters
  <MethodParam name="bool" type="boolean">
    `true` to silence all logs, `false` to resume logging based on other settings.
  </MethodParam>
  
  ### Examples
  <CodeBlock language="javascript">
    {`// Silence all output for production
app.disableLogging(true);`}
  </CodeBlock>
</Method>
<Method
  signature="app.hideTrayIcon(bool)"
  description="Dynamically shows or hides the Novadesk icon in the system tray."
>
  ### Parameters
  <MethodParam name="bool" type="boolean">
    `true` to hide the icon, `false` to show it.
  </MethodParam>
  
  ### Examples
  <CodeBlock language="javascript">
    {`// Run in "stealth" mode
app.hideTrayIcon(true);`}
  </CodeBlock>
</Method>
<Method
  signature="app.useHardwareAcceleration(bool)"
  description="Enables or disables Direct2D hardware acceleration."
>
  <Admonition type="note">
    Changing this setting requires an **application restart** to take effect.
  </Admonition>
  
  ### Parameters
  <MethodParam name="bool" type="boolean">
    `true` to use hardware-accelerated rendering (Default), `false` to use software rendering.
  </MethodParam>
  
  ### Examples
  <CodeBlock language="javascript">
    {`// Enable hardware acceleration
app.useHardwareAcceleration(true);`}
  </CodeBlock>
</Method>
