import CodeBlock from '@theme/CodeBlock';

The `app` object provides methods to control the application's presence in the system tray.

<Admonition type="warning">
Tray methods are **only available in the Main script**. UI scripts cannot modify the system tray.
</Admonition>

<Method
  signature="app.setTrayMenu(items)"
  description="Sets the context menu for the application's system tray icon."
>
  ### Parameters
  <MethodParam name="items" type="Array">
    An array of menu item objects. Each object can have:
    - `text` (string): The text to display for the item.
    - `action` (function): A callback function executed when the item is clicked.
    - `type` (string): If set to `"separator"`, renders a line separator.
    - `checked` (boolean): If `true`, a checkmark is displayed next to the item.
    - `items` (Array): An array of sub-menu items (creates a nested menu).
  </MethodParam>
  
  ### Examples
  <CodeBlock language="javascript">
    {`// index.js
app.setTrayMenu([
  {
    text: "Open App",
    action: function () {
      console.log("Opening...");
    }
  },
  { type: "separator" },
  {
    text: "Settings",
    items: [
      {
        text: "Dark Mode",
        checked: true,
        action: function () {
          toggleDark();
        }
      },
      {
        text: "Launch at Startup",
        checked: false
      }
    ]
  },
  { type: "separator" },
  {
    text: "Exit",
    action: function () {
      app.exit();
    }
  }
]);`}
  </CodeBlock>
</Method>
<Method
  signature="app.clearTrayMenu()"
  description="Removes all custom items from the tray menu."
>
</Method>
<Method
  signature="app.showDefaultTrayItems(show)"
  description="Controls the visibility of default system tray items (like 'Exit')."
>
  ### Parameters
  <MethodParam name="show" type="boolean">
    `true` to show default items, `false` to hide them.
  </MethodParam>
</Method>
<Method
  signature="app.hideTrayIcon(hide)"
  description="Dynamically hides or shows the system tray icon."
>
  ### Parameters
  <MethodParam name="hide" type="boolean">
    `true` to hide the icon, `false` to show it.
  </MethodParam>
  
  ### Examples
  <CodeBlock language="javascript">
    {`app.hideTrayIcon(true);`}
  </CodeBlock>
</Method>
