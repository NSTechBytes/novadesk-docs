---
title: Logging
description: Logging functions via the standard console object.
---

import CodeBlock from '@theme/CodeBlock';
import { Method, MethodParam } from '@site/src/components/Method';
import Admonition from '@theme/Admonition';

<Method
  signature="console.log(...args)"
  description="Log an informational message to the console and log file."
>
  ### Parameters
  <MethodParam name="...args" type="any">
    Values to log.
  </MethodParam>
  
  ### Examples
  <CodeBlock language="javascript">
    {`console.log("Application started");
console.log("User:", "John Doe", "Version:", "1.0.0");`}
  </CodeBlock>
</Method>
<Method
  signature="console.warn(...args)"
  description="Log a warning message to the console and log file."
>
  ### Parameters
  <MethodParam name="...args" type="any">
    Values to log as a warning.
  </MethodParam>
  
  ### Examples
  <CodeBlock language="javascript">
    {`console.warn("Deprecated function used");
console.warn("Low memory warning:", 1024, "MB remaining");`}
  </CodeBlock>
</Method>
<Method
  signature="console.error(...args)"
  description="Log an error message to the console and log file."
>
  ### Parameters
  <MethodParam name="...args" type="any">
    Values to log as an error.
  </MethodParam>
  
  ### Examples
  <CodeBlock language="javascript">
    {`console.error("Failed to load configuration file");
console.error("Error code:", 404, "Message:", "File not found");`}
  </CodeBlock>
</Method>
<Method
  signature="console.debug(...args)"
  description="Log a debug message to the console and log file."
>
  <Admonition type="warning">
    Debug messages are only shown when debug logging is enabled. See how to enable debug logging using [Configuration Methods](/api/app-object/settings).
  </Admonition>
  
  ### Parameters
  <MethodParam name="...args" type="any">
    Values to log as debug information.
  </MethodParam>
  
  ### Examples
  <CodeBlock language="javascript">
    {`console.debug("Debug: Entering function calculatePosition");
console.debug("Debug: Current widget position:", x, y);`}
  </CodeBlock>
</Method>
